33  a=it->first;

36  // Inv: los iteradores posteriors a it no han sido visitados.

38  // a y b son los identificadores que tienen distancia d

40  // Se inicializa el segundo iterador con el valor del segundo elemento del primer iterador

42  it_sec=it->second.begin();
43  // Inv: los iteradores posteriors a it_sec no han sido visitados.
44  // d es el mínimo de los elementos entre it->second.begin y el anterior a it_sec,

110 void Cjt_clusters::actualiza_tab(const string& a, const string& b) {
111  // Actualiza la tabla de distancias
112  map<string,map<string,double> >::iterator it = Tab_clu.begin();
113  string id_fusion=a+b;
114  map<string,double> aux;
115  // Inv: it<=Tab_clu.end()
116  // Se han recalculado y añadido las distancias de los elementos anteriores a it.
117  // Las distancias de los elementos posteriores a it no han sido modificadas.
118  while (it!=Tab_clu.end()) {
119  if (it->first != a and it->first != b) {
120  if (it->first < id_fusion) { // el identificador del clúster fusionado será el segundo identificador del map
121  // recalculo de la distancia del elemento del iterador con el nuevo clúster fusionado
122  double d = (distancia_cl(it->first,a) + distancia_cl(it->first,b))/2;
123  it->second.insert(make_pair(id_fusion,d));
124  }
125  else { // el identificador del clúster fusionado será el primer identificador del map
126  double d = (distancia_cl(a,it->first) + distancia_cl(b,it->first))/2;
127  aux.insert(make_pair(it->first,d));
128  }
129  }
130  ++it;
131  }
132  // Post: se han recalculado todas las distancias de los elementos desde Tab_clu.begin()
133  // hasta Tab_clu.end()-1
134  Tab_clu.erase(Tab_clu.find(a));
135  Tab_clu.erase(Tab_clu.find(b));
136  Tab_clu.insert(make_pair(id_fusion,aux));
137 
138  // Inv: it<=Tab_clu.end(). Se han eliminado los elementos anteriores a it relacionados con a o b.
139  // Las distancias de los elementos posteriores a it no han sido modificadas.
140  // Condición extra para evitar iteraciones innecesarias: el primer elemento de la tabla es menor a a o a b.
141  // Si es mayor, ya no existirán ni a ni b en las distancias.
142  it=Tab_clu.begin();
143  while (it!=Tab_clu.end() and (it->first<a or it->first<b)) {
144  it->second.erase(a);
145  it->second.erase(b);
146  ++it;
147  }
148  // Post: se han eliminado las distancias relacionadas con a y b desde Tab_clu.begin() hasta el
149  // elemento anterior a it.
150 }
